cristo_redentor_decision = {
	is_shown = {
		has_technology_researched = pneumatic_tools
		NOT = {
			any_scope_building = {
				is_building_type = building_cristo_redentor
			}
		}
		NOT = {
			any_scope_state = {
				has_modifier = cristo_redentor_survey
			}
		}
		NOT = {
			has_variable = cristo_redentor_var
		}
		NOT = {
			has_global_variable = cristo_redentor_success_var
		}
	}

	possible = {	
		produced_bureaucracy > 500
		any_scope_state = {
			is_incorporated = yes
			is_split_state = no
		}
		institution_investment_level = {
				institution = institution_social_security
				value >= 5
		}
	}
	
	when_taken = {
		set_variable = {
			name = cristo_redentor_var
			value = 0
		}
		add_modifier = {
			name = cristo_redentor_survey
			months = 12
		}
		add_journal_entry = {
			type = je_cristo_redentor_site
		}
	}

	ai_chance = {
		value = 0
		if = {
			limit = { produced_bureaucracy > 500 }
			add = 5
		}	
		if = {
			limit = { 
				produced_bureaucracy > 5000 
				any_country = {
					is_ai = no
					has_variable = cristo_redentor_var
				}
			}
			add = 50
		}			
		if = {
			limit = { 
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
			}
			add = -1000
		}	
	}
}
