machu_picchu_decision = {
	is_shown = {
		NOT = {
			any_scope_building = {
				is_building_type = building_machu_picchu
			}
		}
		NOT = {
			any_scope_state = {
				has_modifier = machu_picchu_survey
			}
		}
		NOT = {
			has_variable = machu_picchu_var
		}
		NOT = {
			has_global_variable = machu_picchu_success_var
		}
	}

	possible = {	
		produced_bureaucracy > 500
		any_scope_state = {
			is_incorporated = yes
			is_split_state = no
			any_scope_building = {
					is_building_type = building_construction_sector
					level >= 20
			}
		}
	}
	
	when_taken = {
		set_variable = {
			name = machu_picchu_var
			value = 0
		}
		add_modifier = {
			name = machu_picchu_survey
			months = 12
		}
		add_journal_entry = {
			type = je_machu_picchu_site
		}
	}

	ai_chance = {
		value = 0
		if = {
			limit = { produced_bureaucracy > 500 }
			add = 5
		}	
		if = {
			limit = { 
				produced_bureaucracy > 5000 
				any_country = {
					is_ai = no
					has_variable = machu_picchu_var
				}
			}
			add = 50
		}			
		if = {
			limit = { 
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
			}
			add = -1000
		}	
	}
}
