white_house_decision = {
	is_shown = {
		NOT = {
			any_scope_building = {
				is_building_type = building_white_house
			}
		}
		NOT = {
			any_scope_state = {
				has_modifier = white_house_survey
			}
		}
		NOT = {
			has_variable = white_house_var
		}
		NOT = {
			has_global_variable = white_house_success_var
		}
	}

	possible = {	
		produced_bureaucracy > 500
		any_scope_state = {
			is_incorporated = yes
			is_split_state = no
			any_scope_building = {
					is_building_type = building_government_administration
					level >= 20
			}
		}
	}
	
	when_taken = {
		set_variable = {
			name = white_house_var
			value = 0
		}
		add_modifier = {
			name = white_house_survey
			months = 12
		}
		add_journal_entry = {
			type = je_white_house_site
		}
	}

	ai_chance = {
		value = 0
		if = {
			limit = { produced_bureaucracy > 500 }
			add = 5
		}	
		if = {
			limit = { 
				produced_bureaucracy > 5000 
				any_country = {
					is_ai = no
					has_variable = white_house_var
				}
			}
			add = 50
		}			
		if = {
			limit = { 
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
			}			
			add = -1000
		}	
	}
}
