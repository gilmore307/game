central_africa_expedition = { #Nile expedition
	is_shown = {
		NOT = { has_global_variable = central_africa_explored }		
		NOT = { has_variable = on_expedition }	
		OR = {
			is_ai = no
			NOT = { has_variable = ai_expedition_cooldown }
		}		
		institution_investment_level = {
			institution = institution_colonial_affairs
			value >= 3
		}
	}

	possible = {
		is_at_war = no		
		has_interest_marker_in_region = region_zanj
		any_scope_character = {
			OR = {
				has_role = general
				has_role = admiral
			}	
			count >= 3
		}
		institution_investment_level = {
			institution = institution_colonial_affairs
			value >= 3
		}
		calc_true_if = {
			amount >= 1
			owns_entire_state_region = STATE_ZANZIBAR
			owns_entire_state_region = STATE_LINDI
			owns_entire_state_region = STATE_MOCAMBIQUE
			owns_entire_state_region = STATE_TANGANYIKA
			owns_entire_state_region = STATE_SOUTH_MADAGASCAR
			owns_entire_state_region = STATE_NORTH_MADAGASCAR
			owns_entire_state_region = STATE_INDIAN_OCEAN_TERRITORY
			owns_entire_state_region = STATE_KAZEMBE
			owns_entire_state_region = STATE_KENYA
			owns_entire_state_region = STATE_RIFT_VALLEY
			owns_entire_state_region = STATE_UGANDA
		}
	}

	when_taken = {
		if = {
			limit = {
				is_player = yes
			}
			trigger_event = { id = expedition_events.501 popup = yes }
			set_variable = {
				name = on_expedition
				value = yes
			}
		}
		else = {
			set_variable = AI_EXPEDITION_NILE
			ai_expedition_simulator_start = yes
		}
		hidden_effect = {
			set_variable = {
				name = ai_expedition_cooldown
				value = yes
				days = 1825
			}		
		}		
	}
	
	ai_chance = {
		value = 0
		if = {
			limit = {
				weekly_net_fixed_income > money_amount_multiplier_medium
				gold_reserves > 0
			}
			add = 1
		}
		if = {
			limit = {
				has_strategy = ai_strategy_armed_isolationism
			}
			multiply = 0
		}
		if = {
			limit = {
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
			}
			multiply = 0
		}
	}
}
