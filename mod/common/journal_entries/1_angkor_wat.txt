je_angkor_wat_site = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = angkor_wat_var
				add = 1
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
		any_scope_state = {
			is_incorporated = yes
			is_split_state = no
		}
		institution_investment_level = {
			institution = institution_home_affairs
			value >= 5
		}
	}

	on_complete = {
		remove_modifier = angkor_wat_survey
		trigger_event = { id = angkor_wat.1 } # site found
	}

	invalid = {
		NOT = {
			any_scope_state = {
				is_incorporated = yes
				is_split_state = no
			}
			institution_investment_level = {
				institution = institution_home_affairs
				value >= 5
			}
			has_global_variable = angkor_wat_success_var
			any_scope_building = {
				is_building_type = building_angkor_wat
			}
		}
	}

	on_invalid = {
		remove_modifier = angkor_wat_survey
		remove_variable = angkor_wat_var
	}

	current_value = {
		value = root.var:angkor_wat_var
	}

	goal_add_value = {
		value = 1
	}

	progressbar = yes

	weight = 10

	should_be_pinned_by_default = yes
}

je_angkor_wat_construction = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	complete = {
		scope:target = {
			any_scope_building = {
				is_building_type = building_angkor_wat
				is_under_construction = no
			}
		}
	}

	fail = {
		NOT = {
			any_scope_state = {
				has_modifier = angkor_wat_site
			}
		}
	}
	
	on_fail = {
		remove_variable = angkor_wat_var
	}

	on_complete = {
		remove_global_variable = angkor_wat_var
		set_global_variable = angkor_wat_success_var
		scope:target = {
			remove_modifier = angkor_wat_site
		}
	}

	should_be_pinned_by_default = yes
}
