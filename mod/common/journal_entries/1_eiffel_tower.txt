je_eiffel_tower_site = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = eiffel_tower_var
				add = 1
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
		any_scope_state = {
			is_split_state = no
			is_incorporated = yes
			}
		institution_investment_level = {
				institution = institution_workplace_safety
				value >= 5
		}
	}

	on_complete = {
		remove_modifier = eiffel_tower_survey
		trigger_event = { id = eiffel_tower.1 } # site found
	}

	invalid = {
		NOT = {
			any_scope_state = {
				is_split_state = no
				is_incorporated = yes
			}
			institution_investment_level = {
				institution = institution_workplace_safety
				value >= 5
			}
			has_global_variable = eiffel_tower_success_var
			any_scope_building = {
				is_building_type = building_eiffel_tower
			}
		}
	}

	on_invalid = {
		remove_modifier = eiffel_tower_survey
		remove_variable = eiffel_tower_var
	}

	current_value = {
		value = root.var:eiffel_tower_var
	}

	goal_add_value = {
		value = 1
	}

	progressbar = yes

	weight = 10

	should_be_pinned_by_default = yes
}

je_eiffel_tower_construction = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	complete = {
		scope:target = {
			any_scope_building = {
				is_building_type = building_eiffel_tower
				is_under_construction = no
			}
		}
	}

	fail = {
		NOT = {
			any_scope_state = {
				has_modifier = eiffel_tower_site
			}
		}
	}
	
	on_fail = {
		remove_variable = eiffel_tower_var
	}

	on_complete = {
		remove_global_variable = eiffel_tower_var
		set_global_variable = eiffel_tower_success_var
		scope:target = {
			remove_modifier = eiffel_tower_site
		}
	}

	should_be_pinned_by_default = yes
}
