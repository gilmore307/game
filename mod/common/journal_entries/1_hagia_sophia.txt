je_hagia_sophia_site = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = hagia_sophia_var
				add = 1
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
		any_scope_state = {
			is_incorporated = yes
			is_split_state = no
		}
		institution_investment_level = {
			institution = institution_health_system
			value >= 5
		}
	}

	on_complete = {
		remove_modifier = hagia_sophia_survey
		trigger_event = { id = hagia_sophia.1 } # site found
	}

	invalid = {
		NOT = {
			any_scope_state = {
				is_incorporated = yes
				is_split_state = no
			}
			institution_investment_level = {
				institution = institution_health_system
				value >= 5
			}
			has_global_variable = hagia_sophia_success_var
			any_scope_building = {
				is_building_type = building_hagia_sophia
			}
		}
	}

	on_invalid = {
		remove_modifier = hagia_sophia_survey
		remove_variable = hagia_sophia_var
	}

	current_value = {
		value = root.var:hagia_sophia_var
	}

	goal_add_value = {
		value = 1
	}

	progressbar = yes

	weight = 10

	should_be_pinned_by_default = yes
}

je_hagia_sophia_construction = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	complete = {
		scope:target = {
			any_scope_building = {
				is_building_type = building_hagia_sophia
				is_under_construction = no
			}
		}
	}

	fail = {
		NOT = {
			any_scope_state = {
				has_modifier = hagia_sophia_site
			}
		}
	}
	
	on_fail = {
		remove_variable = hagia_sophia_var
	}

	on_complete = {
		remove_global_variable = hagia_sophia_var
		set_global_variable = hagia_sophia_success_var
		scope:target = {
			remove_modifier = hagia_sophia_site
		}
	}

	should_be_pinned_by_default = yes
}
