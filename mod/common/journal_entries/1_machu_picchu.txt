je_machu_picchu_site = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = machu_picchu_var
				add = 1
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
		any_scope_state = {
			is_incorporated = yes
			is_split_state = no
			any_scope_building = {
					is_building_type = building_construction_sector
					level >= 20
			}
		}
	}

	on_complete = {
		remove_modifier = machu_picchu_survey
		trigger_event = { id = machu_picchu.1 } # site found
	}

	invalid = {
		NOT = {
			any_scope_state = {
				is_incorporated = yes
				is_split_state = no
				any_scope_building = {
						is_building_type = building_construction_sector
						level >= 20
				}
			}
			has_global_variable = machu_picchu_success_var
			any_scope_building = {
				is_building_type = building_machu_picchu
			}
		}
	}

	on_invalid = {
		remove_modifier = machu_picchu_survey
		remove_variable = machu_picchu_var
	}

	current_value = {
		value = root.var:machu_picchu_var
	}

	goal_add_value = {
		value = 1
	}

	progressbar = yes

	weight = 10

	should_be_pinned_by_default = yes
}

je_machu_picchu_construction = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	complete = {
		scope:target = {
			any_scope_building = {
				is_building_type = building_machu_picchu
				is_under_construction = no
			}
		}
	}

	fail = {
		NOT = {
			any_scope_state = {
				has_modifier = machu_picchu_site
			}
		}
	}
	
	on_fail = {
		remove_variable = machu_picchu_var
	}

	on_complete = {
		remove_global_variable = machu_picchu_var
		set_global_variable = machu_picchu_success_var
		scope:target = {
			remove_modifier = machu_picchu_site
		}
	}

	should_be_pinned_by_default = yes
}
