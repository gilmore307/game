je_vatican_city_site = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = vatican_city_var
				add = 1
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
		any_scope_state = {
			is_incorporated = yes
			is_split_state = no
			any_scope_building = {
				is_building_type = building_arts_academy
				level >= 20
			}
		}
	}

	on_complete = {
		remove_modifier = vatican_city_survey
		trigger_event = { id = vatican_city.1 } # site found
	}

	invalid = {
		NOT = {
			any_scope_state = {
				is_incorporated = yes
				is_split_state = no
				any_scope_building = {
						is_building_type = building_arts_academy
						level >= 20
				}
			}
			has_global_variable = vatican_city_success_var
			any_scope_building = {
				is_building_type = building_vatican_city
			}
		}
	}

	on_invalid = {
		remove_modifier = vatican_city_survey
		remove_variable = vatican_city_var
	}

	current_value = {
		value = root.var:vatican_city_var
	}

	goal_add_value = {
		value = 1
	}

	progressbar = yes

	weight = 10

	should_be_pinned_by_default = yes
}

je_vatican_city_construction = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_technology

	complete = {
		scope:target = {
			any_scope_building = {
				is_building_type = building_vatican_city
				is_under_construction = no
			}
		}
	}

	fail = {
		NOT = {
			any_scope_state = {
				has_modifier = vatican_city_site
			}
		}
	}
	
	on_fail = {
		remove_variable = vatican_city_var
	}

	on_complete = {
		remove_global_variable = vatican_city_var
		set_global_variable = vatican_city_success_var
		scope:target = {
			remove_modifier = vatican_city_site
		}
	}

	should_be_pinned_by_default = yes
}
