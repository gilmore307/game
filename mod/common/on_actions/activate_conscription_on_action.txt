# Root = State
on_monthly_pulse_state = {
	on_actions = {
		national_guard_conscription_on_action
	}
}

national_guard_conscription_on_action = {
    effect = {
        if = {
            limit = {
                owner = {
                    has_law = law_type:law_national_guard
                }
                has_building = building_conscription_center
			}
            activate_building = {
                building = building_conscription_center
            }
        }
    } 
}

on_monthly_pulse_state = {
	on_actions = {
        secret_police_conscription_on_action
	}
}

secret_police_conscription_on_action = {
    effect = {
        if = {
            limit = {
                owner = {
                    has_law = law_type:law_secret_police
                }
                has_building = building_conscription_center
                has_decree = decree_violent_suppression
			}
            activate_building = {
                building = building_conscription_center
            }
        }
    } 
}
