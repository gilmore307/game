# Root = State
on_monthly_pulse_state = {
	on_actions = {
		claim_check_on_action
	}
}


claim_check_on_action = {
	effect = {
		if = {
			limit = {
				owner = {
					institution_investment_level = {
						institution = institution_colonial_affairs
						value >= 5
					}
					NOT = { has_claim = root.state_region}
				}
				is_incorporated = yes
			} 
			state_region = {
				add_claim = root.owner
			}
		}
	}
}
