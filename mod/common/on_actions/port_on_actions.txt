# Root = State
on_monthly_pulse_state = {
	on_actions = {
		airport_check_on_action
		port_check_on_action
	}
}


airport_check_on_action = {
	effect = {
		if = {
			limit = {
				is_coastal = yes
				has_building = building_airport
			} 
			remove_building = building_airport
			create_building={
 				building = building_port
 				level=1
			}
		}
	}
}

port_check_on_action = {
	effect = {
		if = {
			limit = {
				is_coastal = no
				has_building = building_port
			} 
			remove_building = building_port
			create_building={
 				building = building_airport
 				level=1
			}
		}
	}
}
