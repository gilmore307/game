# Root = State
on_monthly_pulse_state = {
	on_actions = {
		victoria_terminus_on_action
	}
}

victoria_terminus_on_action = {
    effect = {
        if = {
            limit = {
                owner = { has_technology_researched = mass_communication}
            }
            if = {
                limit = {
                    sg:transportation = {
                        state_goods_delta > 50
                    }
                }
                create_building = {
                    building = building_victoria_terminus
                    level = 1
                }
            }
            if = {
                limit = {
                    has_building = building_victoria_terminus
                    or = {
                        sg:transportation = {
                            state_goods_delta < 0
                        }
                    }
                }
                remove_building = building_victoria_terminus
            }
        }
    }
}
